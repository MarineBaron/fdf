## wor contains params for the world
## - zoom, x & y must be between 1 and 100
## - rot must be between -180 and 180

# wor.zoom 50
# wor.x 50
# wor.y 50
# wor.z 50
# wor.rot 0

## cam contains params for the camera position and direction
## - x & y must be between 0 and 199
## - z must be between 0 and 999
## - rotx must be between 90 and 180
## - roty must be between 0 and 180
## - rotz must be between 0 and 180

# cam.x 49
# cam.y 0
# cam.z 999
# cam.rx 90
# cam.ry 90
# cam.rz 90

## pro contains params for the projection
## 1 is orthogonal
## 2 is isometric
## 3 is circulaire

# pro.val 1

## col contains params for the colors
## col.floor.r/g/b are floor's colors composantes
## col.floor.r/g/b are ceil's colors composantes

# col.floor.r 0
# col.floor.g 0
# col.floor.b 0
# col.ceil.r 0
# col.ceil.g 0
# col.ceil.b 0
